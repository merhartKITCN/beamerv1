\begin{frame}{Selections}

	\begin{multicols}{2}
		Muons
		\begin{itemize}
			\item $\pT>\SI{25}{\giga\electronvolt}$
			\item $\abs{\eta}<2.4$
			\item \code[10]{CutBasedIdTight}\footnote{\url{https://twiki.cern.ch/twiki/bin/view/CMS/SWGuideMuonIdRun2}}
			\item \code[10]{PFIsoLoose}
			% \item [\vspace{\fill}]
		\end{itemize}
		\vfill\null
		\columnbreak
		Electrons
		\begin{itemize}
			\item $\pT>\SI{25}{\giga\electronvolt}$
			\item $\abs{\eta}<2.4$
			\item Exclude barrel-endcap transition $1.4442<\abs{\eta}<1.5660$
			\item \code[10]{CutBasedIdTight}\footnote{\url{https://twiki.cern.ch/twiki/bin/viewauth/CMS/CutBasedElectronIdentificationRun2}}
		\end{itemize}	
	\end{multicols}

\end{frame}

\begin{frame}{Selections}

\begin{frame}{Selections}
